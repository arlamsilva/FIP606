p_box <- mg |> 
  ggplot(aes(trat, comp))+
  #geom_point()+
  geom_boxplot(outlier.color = NA,
               fill = "orange",
               size = 0.5,
               width = 0.4)+
  geom_jitter(width = 0.1, 
              height = 0,
              size =2,
              color = "black")+
  scale_y_continuous(limits = c(7,19),
                     n.breaks = 6)+
  labs(y = "Lesion size (mm)",
       x = " ")+
  theme_bw()

library(ggthemes)
Warning: package 'ggthemes' was built under R version 4.2.3
p_means <- mg |> 
  group_by(trat) |> 
  summarise(comp_mean = mean(comp),
            comp_sd = sd(comp)) |> 
  ggplot(aes(trat, comp_mean))+
  #geom_col(fill = "orange",
   #        width = 0.5)+
  geom_point()+
  scale_y_continuous(limits = c(7,18),
                     n.breaks = 6)+
  geom_errorbar(aes(ymin = comp_mean - comp_sd , 
                    ymax = comp_mean + comp_sd,
                    width = 0.05))+
  theme_bw()+
  labs(y = "Lesion size (mm)",
       x = " ")
p_means

       height = 4,
